输出大于所输入数的素数

#include <stdio.h>
#include <stdlib.h>

int is_prime(int n) {
    if (n <= 1) {
        return 0;
    }
    for (int i = 2; i * i <= n; i++) {
        if (n % i == 0) {
            return 0;
        }
    }
    return 1;
}

int main() {
    int n, count = 0;
    printf("请输入一个整数：");
    scanf("%d", &n);
    for (int i = n + 1; count < 3; i++) {
        if (is_prime(i)) {
            printf("%d ", i);
            count++;
        }
    }
    return 0;
}

用C语言实现整数的大小排序

#include <stdio.h>


int main() {
    int a, b, c, t;
    printf("请输入三个整数，用逗号隔开：");
    scanf("%d,%d,%d", &a, &b, &c);
    if (a > b) {
        t = a;
        a = b;
        b = t;
    }
    if (a > c) {
        t = a;
        a = c;
        c = t;
    }
    if (b > c) {
        t = b;
        b = c;
        c = t;
    }
    printf("排序后的结果为：%d, %d, %d\n", a, b, c);
    return 0;
}

判断三角形

#include <stdio.h>
#include <math.h>

int main() {
    float a, b, c;
    printf("请输入三角形的三条边，用逗号隔开：");
    scanf("%f,%f,%f", &a, &b, &c);
    if (a + b > c && b + c > a && a + c > b) {
        printf("这是一个");
        if (a == b && b == c) {
            printf("等边三角形");
        } else if (a == b || b == c || a == c) {
            printf("等腰三角形");
        } else if (pow(a, 2) + pow(b, 2) == pow(c, 2) || pow(a, 2) + pow(c, 2) == pow(b, 2) || pow(b, 2) + pow(c, 2) == pow(a, 2)) {
            printf("直角三角形");
        } else {
            printf("普通三角形");
        }
        printf("\n");
    } else {
        printf("这不是一个三角形\n");
    }
    return 0;
}

int main() {
    int n, i, j, k;
    printf("请输入菱形边长（奇数）：");
    scanf("%d", &n);
    for (i = 1; i <= n; ++i) {
        for (j = 1; j <= n - i; ++j) {
            printf(" ");
        }
        for (k = 1; k <= 2 * i - 1; ++k) {
            printf("*");
        }
        printf("\n");
    }
    for (i = n - 1; i >= 1; --i) {
        for (j = 1; j <= n - i; ++j) {
            printf(" ");
        }
        for (k = 1; k <= 2 * i - 1; ++k) {
            printf("*");
        }
        printf("\n");
    }
    return 0;
}

递归方式的斐波那契数列

#include <stdio.h>

int fibonacci(int n) {
    if (n == 0 || n == 1) {
        return n;
    } else {
        return fibonacci(n - 1) + fibonacci(n - 2);
    }
}

int main() {
    int n, i;
    printf("请输入要计算的斐波那契数列的项数：");
    scanf("%d", &n);
    printf("斐波那契数列的前 %d 项为：", n);
    for (i = 0; i < n; ++i) {
        printf("%d ", fibonacci(i));
    }
    printf("\n");
    return 0;
}

计算公式p=m!/n!(m-n)!

#include <stdio.h>

int main() {
    int m, n, i;
    double p = 1;
    printf("请输入m和n的值，用逗号隔开：");
    scanf("%d,%d", &m, &n);
    if (m < n) {
        printf("m必须大于等于n！\n");
        return 0;
    }
    for (i = 1; i <= n; ++i) {
        p *= (m - i + 1) * 1.0 / i;
    }
    printf("p的值为：%lf\n", p);
    return 0;
}

void reverse_printf(const char *str)
{
    if(str==NULL || *str=='\0')
    {
        return;
    }
    reverse_printf(str+1);

}

转换字符大小写

#include <stdio.h>
#include <string.h>

int main() {
    char str[100];
    int i;
    printf("请输入一个字符串：");
    scanf("%s", str);
    for (i = 0; str[i] != '\0'; ++i) {
        if (str[i] >= 'a' && str[i] <= 'z') {
            str[i] = str[i] & 0xDF;
        } else if (str[i] >= 'A' && str[i] <= 'Z') {
            str[i] = str[i] | 0x20;
        }
    }
    printf("转换后的字符串为：%s\n", str);
    return 0;
}

冒泡法排序

#include <stdio.h>

int main() {
    int a[] = {900, 2, 3, -58, 34, 76, 32, 43, 56, -70, 35, -234, 532, 543, 2500};
    int n = sizeof(a) / sizeof(int);
    int i, j, buf;
    for (i = 0; i < n - 1; ++i) {
        for (j = 0; j < n - 1 - i; ++j) {
            if (a[j] < a[j + 1]) {
                buf = a[j];
                a[j] = a[j + 1];
		        a[j + 1] = buf;
            }
        }
    }
    printf("排序后的结果为：");
    for (i = 0; i < n; ++i) {
        printf("%d ", a[i]);
    }
    printf("\n");
    return 0;
}

void bubble_sort(int arr[], int n) { 
    for(int i = 0; i < n-1; i++); {
        for(int j = 0; j < n-i-1; j++) {
            if(arr[j] > arr[j+1]) {
                // 交换 arr[j] 和 arr[j+1]
                int temp = arr[j];
                arr[j] = arr[j+1];
                arr[j+1] = temp;
            }
        }
    }
}

打印杨辉三角

#include <stdio.h>

int main() {
    int n, i, j;
    printf("请输入要打印的行数：");
    scanf("%d", &n);
    int a[n][n];
    for (i = 0; i < n; ++i) {
        for (j = 0; j <= i; ++j) {
            if (j == 0 || j == i) {
                a[i][j] = 1;
            } else {
                a[i][j] = a[i - 1][j - 1] + a[i - 1][j];
            }
            printf("%d ", a[i][j]);
        }
        printf("\n");
    }
    return 0;
}
