#include <stdio.h>
#include <string.h>
#include <stdlib.h>

typedef struct _student
{
    int score;
    char name[16];
}student,*pstudent;

int compare_student_score1(student s1, student s2)
{
    return s1.score - s2.score;
}

int compare_student_score2(student* s1, student* s2)
{
    return s1->score - s2->score;
}

int main() 
{
    student stu1[3] = { {78,"tom"}, {83,"lily"}, {79,"david"} };
    student stu2[3] = { 0 };
    pstudent stu3[3] = { 0 };
    
    for (int i = 0; i < 3; i++)
    {
        printf("Please input score and name for %d\n", i + 1);
        scanf_s("%d%s", &stu2[i].score, stu2[i].name, 16);
    }
    
    for (int i = 0; i < 3; i++)
    {
        stu3[i] = &stu1[i];
    }
    
    for (int i = 0; i < 3; i++)
    {
        printf("stu1:%d:score:%d,name:%s\n", i, stu1[i].score,
            stu1[i].name);
    }
    
    for (int i = 0; i < 3; i++)
    {
        printf("stu2:%d:score:%d,name:%s\n", i, stu2[i].score,
            stu2[i].name);
    }
    
    for (int i = 0; i < 3; i++)
    {
        printf("stu3:%d:score:%d,name:%s\n",i,stu3[i]->score,
            stu3[i]->name);
    }
    
	student s3 = { 85,"a" };
    student s4 = { 91,"b" };
    
    int res = compare_student_score1(s3,s4);
    
    if (res == 0)
    {
        printf("the score of %s equals the score of %s\n",
            s3.name,s4.name);
    }
    
    else if (res > 0)
    {
        printf("the score of %s above the score of %s\n",
            s3.name, s4.name);
    }
    
    else
    {
        printf("the score of %s below the score of %s\n",
            s3.name, s4.name);
    }
    
    return 0;
}
